## 강지웅
```swift
import Foundation

func solution(_ numbers:[Int]) -> Int {
    var result: Int = 0
    var isContains = [Bool](repeating: false, count: 10)
    
    for number in numbers {
        isContains[number] = true
    }
    
    for i in 0..<10 {
        if !isContains[i] {
            result += i
        }
    }
    
    return result
}
```
