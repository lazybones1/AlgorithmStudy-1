## 정윤영
```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.LinkedList;
import java.util.Queue;

public class Main {

    static char[][] arr = new char[5][5];
    static int cnt;
    static ArrayList<Integer> result = new ArrayList<>();

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        for(int i=0; i<5; i++){
            String str = br.readLine();
            for(int j=0; j<5; j++){
                arr[i][j] = str.charAt(j);
            }
        }

        comb(0, 0);

        System.out.println(cnt);
    }

    //칠공주 조합
    public static void comb(int start, int r){
        if(r == 7){
            if(bfs() && checkCnt()){
                cnt++;
            }
            return;
        }

        for(int i=start; i<25; i++){
            result.add(i);
            comb(i+1, r+1);
            result.remove(result.size()-1);
        }
    }

    //칠공주 연결 여부 확인
    public static boolean bfs(){
        Queue<int[]> queue = new LinkedList<>();
        int[][] deltas = {{1,0}, {-1,0}, {0,-1}, {0,1}};
        int count = 0;
        boolean[][] isVisited = new boolean[5][5];

        int r = result.get(0)/5;
        int c = result.get(0)%5;
        queue.add(new int[]{r, c});
        isVisited[r][c] = true;
        count++;

        while(!queue.isEmpty()){
            if(count == 7){
                return true;
            }
            int[] cur = queue.poll();
            for(int i=0; i<4; i++){
                int dr = cur[0] + deltas[i][0];
                int dc = cur[1] + deltas[i][1];

                if(checkBoundary(dr, dc) && !isVisited[dr][dc] && result.contains(dr*5+dc)){
                    queue.add(new int[]{dr, dc});
                    count++;
                    isVisited[dr][dc] = true;
                }
            }
        }
        return false;
    }

    public static boolean checkBoundary(int dr, int dc){
        return dr>=0 && dc>=0 && dr<5 && dc<5;
    }

    //이다솜파가 더 많으면 true
    public static boolean checkCnt(){
        int cntS = 0;

        for(int pos : result){
            int r = pos/5;
            int c = pos%5;

            if(arr[r][c] == 'S'){
                cntS++;
            }
        }

        if(cntS >= 4){
            return true;
        }else {
            return false;
        }
    }
}
```
