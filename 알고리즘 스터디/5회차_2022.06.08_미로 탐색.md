## 강지웅
```swift
func mazes() {
    let dx: [Int] = [0, 0, -1, 1]
    let dy: [Int] = [1, -1, 0, 0]
    
    let input: [Int] = readLine()!.split(separator: " ").map{ Int($0)! }
    let N: Int = input[0]
    let M: Int = input[1]
    
    var point: [Int]
    var next: [Int]
    var queue: [[Int]] = [[0, 0, 0]]
    var maze: [[Character]] = []
    
    for _ in 0..<N {
        maze.append(readLine()!.map{$0})
    }
    
    maze[0][0] = "2"
    
    while(!queue.isEmpty){
        point = queue.removeFirst()
        point[2] += 1
        
        if point[0] == N - 1 && point[1] == M - 1{
            print("\(point[2])")
        }
        
        for i in 0..<4 {
            next = [point[0] + dx[i], point[1] + dy[i], point[2]]
            if next[0] < 0 || next[0] >= N || next[1] < 0 || next[1] >= M {
                continue
            }
            if maze[next[0]][next[1]] == "1"{
                queue.append(next)
                maze[next[0]][next[1]] = "2"
            }
        }
    }
}
```
