## 오나연
```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.LinkedList;
import java.util.List;
import java.util.Queue;
import java.util.StringTokenizer;

public class Main_BOJ_2589_보물섬_0428 {
	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st = new StringTokenizer(br.readLine());
		
		int N = Integer.parseInt(st.nextToken());
		int M = Integer.parseInt(st.nextToken());
		
		char[][] map = new char[N][M];
		List<Loc> list = new ArrayList<>();
		
		for(int i=0; i<N; i++) {
			String tmp = br.readLine();
			for(int j=0; j<M; j++) {
				map[i][j] = tmp.charAt(j);
				if(map[i][j] == 'L') list.add(new Loc(i, j, 1));
			}
		}
		
		int size = list.size();
		int max = 0;
		for(int i=0; i<size; i++) {
			Loc loc = list.get(i);
			int time = bfs(map, loc, N, M);
			max = (time>max)? time: max;
		}
		System.out.println(max);
	}
	
	public static int bfs(char[][] map, Loc loc, int N, int M) {
		int[] dx = {-1, 0, 1, 0};
		int[] dy = {0, 1, 0, -1};
		
		Queue<Loc> q = new LinkedList<>();
		q.offer(loc);
		int time = 0;
		boolean[][] visited = new boolean[N][M];
		visited[loc.x][loc.y] = true;
		
		while(!q.isEmpty()) {
			Loc cur = q.poll();
			int x = cur.x;
			int y = cur.y;
			time = cur.time;
			
			for(int dir=0; dir<4; dir++) {
				int nextX = x + dx[dir];
				int nextY = y + dy[dir];
				if(nextX<0 || nextX>=N || nextY<0 || nextY>=M) continue;
				if(map[nextX][nextY] == 'W') continue;
				if(visited[nextX][nextY]) continue;
				
				visited[nextX][nextY] = true;
				q.offer(new Loc(nextX, nextY, time+1));
			}
		}
		return time-1;
	}
	
	static class Loc{
		int x, y, time;
		public Loc(int x, int y, int time) {
			this.x = x;
			this.y = y;
			this.time = time;
		}
	}
}
```

## 정윤영
```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.lang.reflect.Array;
import java.util.Arrays;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class Main {

    static char[][] map;
    static int N, M, max;
    static int[][] distance;
    static int[][] deltas = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        StringTokenizer st = new StringTokenizer(br.readLine());
        N = Integer.parseInt(st.nextToken());
        M = Integer.parseInt(st.nextToken());

        map = new char[N][M];
        for(int i=0; i<N; i++){
            String str = br.readLine();
            for(int j=0; j<M; j++){
                map[i][j] = str.charAt(j);
            }
        }

        for(int i=0; i<N; i++){
            for(int j=0; j<M; j++){
                if(map[i][j] == 'L'){
                    distance = new int[N][M];

                    for(int[] temp : distance){
                        Arrays.fill(temp, -1);
                    }
                    
                    bfs(i, j);
                }
            }
        }

        System.out.println(max);
    }

    public static void bfs(int i, int j){
        Queue<int[]> queue = new LinkedList<>();
        queue.add(new int[]{i, j});
        distance[i][j]++;
        int dis = 0;

        while(!queue.isEmpty()){
            int[] temp = queue.poll();
            int r = temp[0];
            int c = temp[1];
            dis = distance[r][c];

            for(int d=0; d<4; d++){
                int dr = r + deltas[d][0];
                int dc = c + deltas[d][1];
                if(checkBoundary(dr, dc) && map[dr][dc] == 'L' && distance[dr][dc] == -1){
                    queue.add(new int[]{dr, dc});
                    distance[dr][dc] = dis+1;
                }
            }
        }
        max = Math.max(max, dis);
    }

    public static boolean checkBoundary(int dr, int dc){
        return dr>=0 && dc>=0 && dr<N && dc<M;
    }
}
```
