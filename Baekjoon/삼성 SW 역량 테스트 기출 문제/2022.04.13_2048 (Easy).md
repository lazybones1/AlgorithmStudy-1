## 이주형
```java
import java.util.Scanner;
import java.util.Stack;

public class Main {
    private static int n;
    private static int maxScore;

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        n = sc.nextInt();
        int[][] board = new int[n][n];
        maxScore = 0;

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                board[i][j] = sc.nextInt();

                if(board[i][j] > maxScore) {
                    maxScore = board[i][j];
                }
            }
        }

        playGame(0, board);
        System.out.println(maxScore);
    }

    private static void playGame(int turnCount, int[][] nextBoard) {
        if (turnCount == 5) {
            return;
        }

        playGame(turnCount + 1, moveUp(nextBoard));
        playGame(turnCount + 1, moveDown(nextBoard));
        playGame(turnCount + 1, moveRight(nextBoard));
        playGame(turnCount + 1, moveLeft(nextBoard));
    }

    private static int[][] moveUp(int[][] currentBoard) {
        int[][] nextBoard = new int[n][n];
        Stack<Integer> stack = new Stack<>();
        boolean isMerged = false;

        for (int j = 0; j < n; j++) {
            for (int i = 0; i < n; i++) {
                if (currentBoard[i][j] != 0) {
                   if (stack.size() == 0) {
                       stack.add(currentBoard[i][j]);
                   } else {
                       if (stack.peek() == currentBoard[i][j] && !isMerged)  {
                           stack.add(stack.pop() + currentBoard[i][j]);
                           isMerged = true;

                           if(stack.peek() > maxScore) {
                               maxScore = stack.peek();
                           }

                           continue;
                       } else {
                           stack.add(currentBoard[i][j]);
                       }
                   }
                   isMerged = false;
                }
            }
            while (!stack.isEmpty()) {
                nextBoard[stack.size() - 1][j] = stack.pop();
            }
        }
        return nextBoard;
    }


    private static int[][] moveDown(int[][] currentBoard) {
        int[][] nextBoard = new int[n][n];
        Stack<Integer> stack = new Stack<>();
        boolean isMerged = false;

        for (int j = 0; j < n; j++) {
            for (int i = n - 1; i >= n; i--) {
                if (currentBoard[i][j] != 0) {
                    if (stack.size() == 0) {
                        stack.add(currentBoard[i][j]);
                    } else {
                        if (stack.peek() == currentBoard[i][j] && !isMerged)  {
                            stack.add(stack.pop() + currentBoard[i][j]);
                            isMerged = true;

                            if(stack.peek() > maxScore) {
                                maxScore = stack.peek();
                            }

                            continue;
                        } else {
                            stack.add(currentBoard[i][j]);
                        }
                    }
                    isMerged = false;
                }
            }
            while (!stack.isEmpty()) {
                nextBoard[n - stack.size()][j] = stack.pop();
            }
        }
        return nextBoard;
    }

    private static int[][] moveRight(int[][] currentBoard) {
        int[][] nextBoard = new int[n][n];
        Stack<Integer> stack = new Stack<>();
        boolean isMerged = false;

        for (int i = 0; i < n; i++) {
            for (int j = n - 1; j >= 0; j--) {
                if (currentBoard[i][j] != 0) {
                    if (stack.size() == 0) {
                        stack.add(currentBoard[i][j]);
                    } else {
                        if (stack.peek() == currentBoard[i][j] && !isMerged)  {
                            stack.add(stack.pop() + currentBoard[i][j]);
                            isMerged = true;

                            if(stack.peek() > maxScore) {
                                maxScore = stack.peek();
                            }

                            continue;
                        } else {
                            stack.add(currentBoard[i][j]);
                        }
                    }
                    isMerged = false;
                }
            }
            while (!stack.isEmpty()) {
                nextBoard[i][n - stack.size()] = stack.pop();
            }
        }
        return nextBoard;
    }

    private static int[][] moveLeft(int[][] currentBoard) {
        int[][] nextBoard = new int[n][n];
        Stack<Integer> stack = new Stack<>();
        boolean isMerged = false;

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (currentBoard[i][j] != 0) {
                    if (stack.size() == 0) {
                        stack.add(currentBoard[i][j]);
                    } else {
                        if (stack.peek() == currentBoard[i][j] && !isMerged) {
                            stack.add(stack.pop() + currentBoard[i][j]);
                            isMerged = true;

                            if(stack.peek() > maxScore) {
                                maxScore = stack.peek();
                            }

                            continue;
                        } else {
                            stack.add(currentBoard[i][j]);
                        }
                    }
                    isMerged = false;
                }
            }
            while (!stack.isEmpty()) {
                nextBoard[i][stack.size() - 1] = stack.pop();
            }
        }
        return nextBoard;
    }
}
```

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class BOJ_12100 {

//    static int[][] deltas = {{1,0}, {-1, 0}, {0,1}, {0,-1}};
    static int max, N;
    static int[][] arr;
    static int[] result = new int[5];

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        N = Integer.parseInt(br.readLine());

        arr = new int[N][N];
        for(int i=0; i<N; i++){
            StringTokenizer st = new StringTokenizer(br.readLine());
            for(int j=0; j<N; j++){
                arr[i][j] = Integer.parseInt(st.nextToken());
            }
        }

        dfs(0);

        System.out.println(max);
    }

    public static void dfs(int depth){
        if(depth == 5){
            int[][] temp = move();
            max = Math.max(max, findMax(temp));
            return;
        }

        for(int i=0; i<4; i++){
            result[depth] = i;
            dfs(depth+1);
        }
    }

    public static int[][] move(){
        int[][] temp = new int[N][N];
        cloneArr(temp);


        //방향에 맞춰서 큐에 넣고 연속적인 값이 같으면 더해서 방향에 맞춰 배열에 넣기
        for(int i=0; i<5; i++){     // 정해진 방향에 맞춰서 이동
            switch (result[i]){
                case 0 :        //상 (세로별로 확인)
                    for(int j=0; j<N; j++){
                        Queue<Integer> queue = new LinkedList<>();
                        for(int k=0; k<N; k++){
                            if(temp[k][j] != 0){
                                queue.add(temp[k][j]);
                                temp[k][j] = 0;
                            }
                        }
                        if(queue.isEmpty()){
                            continue;
                        }
                        int pre = queue.poll();
                        int idx = 0;
                        while (!queue.isEmpty()){
                            int cur = queue.poll();
                            if(pre == cur){
                                temp[idx++][j] = pre*2;
                                if(queue.isEmpty()){
                                    pre = 0;
                                    break;
                                }else{
                                    pre = queue.poll();
                                }
                            }else{
                                temp[idx++][j] = pre;
                                pre = cur;
                            }
                        }
                        temp[idx][j] = pre;
                    }
                    break;
                case 1:         //하
                    for(int j=0; j<N; j++){
                        Queue<Integer> queue = new LinkedList<>();
                        for(int k=N-1; k>=0; k--){
                            if(temp[k][j] != 0){
                                queue.add(temp[k][j]);
                                temp[k][j] = 0;
                            }
                        }
                        if(queue.isEmpty()){
                            continue;
                        }
                        int pre = queue.poll();
                        int idx = N-1;
                        while (!queue.isEmpty()){
                            int cur = queue.poll();
                            if(pre == cur){
                                temp[idx--][j] = pre*2;
                                if(queue.isEmpty()){
                                    pre = 0;
                                    break;
                                }else{
                                    pre = queue.poll();
                                }
                            }else{
                                temp[idx--][j] = pre;
                                pre = cur;
                            }
                        }
                        temp[idx][j] = pre;
                    }
                    break;

                case 2:         //좌
                    for(int j=0; j<N; j++){
                        Queue<Integer> queue = new LinkedList<>();
                        for(int k=0; k<N; k++){
                            if(temp[j][k] != 0){
                                queue.add(temp[j][k]);
                                temp[j][k] = 0;
                            }
                        }
                        if(queue.isEmpty()){
                            continue;
                        }
                        int pre = queue.poll();
                        int idx = 0;
                        while (!queue.isEmpty()){
                            int cur = queue.poll();
                            if(pre == cur){
                                temp[j][idx++] = pre*2;
                                if(queue.isEmpty()){
                                    pre = 0;
                                    break;
                                }else{
                                    pre = queue.poll();
                                }
                            }else{
                                temp[j][idx++] = pre;
                                pre = cur;
                            }
                        }
                        temp[j][idx] = pre;
                    }
                    break;

                case 3:         //우
                    for(int j=0; j<N; j++){
                        Queue<Integer> queue = new LinkedList<>();
                        for(int k=N-1; k>=0; k--){
                            if(temp[j][k] != 0){
                                queue.add(temp[j][k]);
                                temp[j][k] = 0;
                            }
                        }
                        if(queue.isEmpty()){
                            continue;
                        }
                        int pre = queue.poll();
                        int idx = N-1;
                        while (!queue.isEmpty()){
                            int cur = queue.poll();
                            if(pre == cur){
                                temp[j][idx--] = pre*2;
                                if(queue.isEmpty()){
                                    pre = 0;
                                    break;
                                }else{
                                    pre = queue.poll();
                                }
                            }else{
                                temp[j][idx--] = pre;
                                pre = cur;
                            }
                        }
                        temp[j][idx] = pre;
                    }
                    break;
            }
        }
        return temp;
    }

    public static void cloneArr(int[][] temp){
        for(int i=0; i<N; i++){
            for(int j=0; j<N; j++){
                temp[i][j] = arr[i][j];
            }
        }
    }

    public static int findMax(int[][] temp){
        int x = 0;
        for(int i=0; i<N; i++){
            for(int j=0; j<N; j++){
                x = Math.max(x, temp[i][j]);
            }
        }

        return x;
    }
}
```
