import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class BOJ_12100 {

//    static int[][] deltas = {{1,0}, {-1, 0}, {0,1}, {0,-1}};
    static int max, N;
    static int[][] arr;
    static int[] result = new int[5];

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        N = Integer.parseInt(br.readLine());

        arr = new int[N][N];
        for(int i=0; i<N; i++){
            StringTokenizer st = new StringTokenizer(br.readLine());
            for(int j=0; j<N; j++){
                arr[i][j] = Integer.parseInt(st.nextToken());
            }
        }

        dfs(0);

        System.out.println(max);
    }

    public static void dfs(int depth){
        if(depth == 5){
            int[][] temp = move();
            max = Math.max(max, findMax(temp));
            return;
        }

        for(int i=0; i<4; i++){
            result[depth] = i;
            dfs(depth+1);
        }
    }

    public static int[][] move(){
        int[][] temp = new int[N][N];
        cloneArr(temp);


        //방향에 맞춰서 큐에 넣고 연속적인 값이 같으면 더해서 방향에 맞춰 배열에 넣기
        for(int i=0; i<5; i++){     // 정해진 방향에 맞춰서 이동
            switch (result[i]){
                case 0 :        //상 (세로별로 확인)
                    for(int j=0; j<N; j++){
                        Queue<Integer> queue = new LinkedList<>();
                        for(int k=0; k<N; k++){
                            if(temp[k][j] != 0){
                                queue.add(temp[k][j]);
                                temp[k][j] = 0;
                            }
                        }
                        if(queue.isEmpty()){
                            continue;
                        }
                        int pre = queue.poll();
                        int idx = 0;
                        while (!queue.isEmpty()){
                            int cur = queue.poll();
                            if(pre == cur){
                                temp[idx++][j] = pre*2;
                                if(queue.isEmpty()){
                                    pre = 0;
                                    break;
                                }else{
                                    pre = queue.poll();
                                }
                            }else{
                                temp[idx++][j] = pre;
                                pre = cur;
                            }
                        }
                        temp[idx][j] = pre;
                    }
                    break;
                case 1:         //하
                    for(int j=0; j<N; j++){
                        Queue<Integer> queue = new LinkedList<>();
                        for(int k=N-1; k>=0; k--){
                            if(temp[k][j] != 0){
                                queue.add(temp[k][j]);
                                temp[k][j] = 0;
                            }
                        }
                        if(queue.isEmpty()){
                            continue;
                        }
                        int pre = queue.poll();
                        int idx = N-1;
                        while (!queue.isEmpty()){
                            int cur = queue.poll();
                            if(pre == cur){
                                temp[idx--][j] = pre*2;
                                if(queue.isEmpty()){
                                    pre = 0;
                                    break;
                                }else{
                                    pre = queue.poll();
                                }
                            }else{
                                temp[idx--][j] = pre;
                                pre = cur;
                            }
                        }
                        temp[idx][j] = pre;
                    }
                    break;

                case 2:         //좌
                    for(int j=0; j<N; j++){
                        Queue<Integer> queue = new LinkedList<>();
                        for(int k=0; k<N; k++){
                            if(temp[j][k] != 0){
                                queue.add(temp[j][k]);
                                temp[j][k] = 0;
                            }
                        }
                        if(queue.isEmpty()){
                            continue;
                        }
                        int pre = queue.poll();
                        int idx = 0;
                        while (!queue.isEmpty()){
                            int cur = queue.poll();
                            if(pre == cur){
                                temp[j][idx++] = pre*2;
                                if(queue.isEmpty()){
                                    pre = 0;
                                    break;
                                }else{
                                    pre = queue.poll();
                                }
                            }else{
                                temp[j][idx++] = pre;
                                pre = cur;
                            }
                        }
                        temp[j][idx] = pre;
                    }
                    break;

                case 3:         //우
                    for(int j=0; j<N; j++){
                        Queue<Integer> queue = new LinkedList<>();
                        for(int k=N-1; k>=0; k--){
                            if(temp[j][k] != 0){
                                queue.add(temp[j][k]);
                                temp[j][k] = 0;
                            }
                        }
                        if(queue.isEmpty()){
                            continue;
                        }
                        int pre = queue.poll();
                        int idx = N-1;
                        while (!queue.isEmpty()){
                            int cur = queue.poll();
                            if(pre == cur){
                                temp[j][idx--] = pre*2;
                                if(queue.isEmpty()){
                                    pre = 0;
                                    break;
                                }else{
                                    pre = queue.poll();
                                }
                            }else{
                                temp[j][idx--] = pre;
                                pre = cur;
                            }
                        }
                        temp[j][idx] = pre;
                    }
                    break;
            }
        }
        return temp;
    }

    public static void cloneArr(int[][] temp){
        for(int i=0; i<N; i++){
            for(int j=0; j<N; j++){
                temp[i][j] = arr[i][j];
            }
        }
    }

    public static int findMax(int[][] temp){
        int x = 0;
        for(int i=0; i<N; i++){
            for(int j=0; j<N; j++){
                x = Math.max(x, temp[i][j]);
            }
        }

        return x;
    }
}
